% resume_templates.tex

% Define global length variables for paragraph and section spacing
\newlength{\parskiplength}
\newlength{\sectionskiplength}

% Custom Hyper Link Skill
\newcommand{\customlink}[2]{
    \href{#1}{\underline{#2}}
}

% Custom Header Command Template
\newcommand{\customheader}[4]{
    \begin{center}
        \textbf{\Large #1} \\ 
        \textbf{Phone:} #2 \, 
        \textbf{Email:} \customlink{mailto:#3}{#3} \, 
        \textbf{LinkedIn:} \customlink{https://www.#4}{#4}
    \end{center}
    \vspace{-1em} % Reduce space after the header
}

% Base custom section environment with adjustable paragraph spacing
\newenvironment{basecustomsection}[2]{
    \vspace{-1.5em + \sectionskiplength}
    \section*{\normalsize \MakeUppercase{#1}} % Section header 
    \vspace{-0.75em} % Reduce space between header and line
    \hrule
    \begingroup
    \setlength{\parindent}{0em} % No indentation for paragraphs
    \setlength{\parskip}{#2}
}{
    \endgroup
}

% Custom section with paragraph spacing
\newenvironment{customsection}[1]{
    \begin{basecustomsection}{#1}{\parskiplength}
}{
    \end{basecustomsection}
}

% Custom section without paragraph spacing
\newenvironment{customsectionnoskip}[1]{
    \begin{basecustomsection}{#1}{0em}
    \vspace{\parskiplength}
}{
    \end{basecustomsection}
}

% Custom itemize environment with zero spacing between items
\newenvironment{customitemize}{
    \vspace{-\parskip}
    \begin{itemize}[left=0.5em, topsep=0em, itemsep=0em, partopsep=0em, parsep=0em]
}{
    \end{itemize}
}

% Skills Template
\newcommand{\skillgroup}[2]{
    \textbf{#1}: #2. 
}

% Experience Template
\newcommand{\experience}[6]{
    \textbf{#1}, #2. \hfill #3 \\
    \textit{#4}
    \ifstrempty{#5}{}{\\ \textit{Tech Stack: #5.}} % Conditional for Tech Stack
    \begin{customitemize}
        #6 % This will allow you to use \item inside the command
    \end{customitemize}
}

% Project Template
\newcommand{\project}[5]{
    \textbf{#1}. #2. \hfill #3
    \ifstrempty{#4}{}{\\ \textit{Tech Stack: #4.}} % Conditional for Tech Stack
    \begin{customitemize}
        #5 % This will allow you to use \item inside the command
    \end{customitemize}
}

% Education Template
\newcommand{\education}[4]{
    \textbf{#1}, #2. \hfill #3 \\ 
    #4. 
}

% Award Template
